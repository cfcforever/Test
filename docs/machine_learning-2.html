using the model

data$consec = NA
data$stab = NA
for (k in 1:(nrow(data)-9)){
  td = data$timediff[k:(k+9)]
  if (all(td>=0.19) & all(td<=0.21)){
    data$consec[k] = T
    temp = as.numeric(dist(data[k:(k+9), c("x","y")]))
    data$stab[k] = predict(ir, temp)
  }else{
    data$consec[k] = F
  }
}

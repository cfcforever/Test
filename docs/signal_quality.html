for (k in 1:nb_tag){
  tag = tagId[k]
  dat_tag <- list_tag[tag][[1]]
  dat_tag$diff_ts = c(dat_tag$record_timestamp[-1]-dat_tag$record_timestamp[-nrow(dat_tag)],0)
  list_tag[tag][[1]] = dat_tag
}

dat_lowq = do.call(rbind.data.frame, list_tag)

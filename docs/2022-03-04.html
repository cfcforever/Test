for (k in 1:nb_tag){
  tag = tagId[k]
  dat_tag = list_tag[tag][[1]]
  
  rows_sel = which(dat_tag$diff_ts>5)
  if (length(rows_sel)>=1){
    nr = c()
    nd = c()
    ns = c()
    for (i in 1:length(rows_sel)){
      ki = rows_sel[i]
      s = dat_tag[(ki-1):ki,c("x","y")]
      if (sum(is.na(s))==0){
        nr = c(nr, ki)
        ns = c(ns, dat_tag$diff_ts[ki])
        d = dist(s)/100
        nd = c(nd, d)
      }
